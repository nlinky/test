git

copy NUL .gitconfig

git config --list --global

cd папка

git clone адрес репозитория

cd новая папка

git status
>красным

git add имя файла
>зеленым
>после изменнения снова эта команда

git commit -m "test commit 1"
> коммитим, название на сервере, но пока только у нас на локал

git push
>пушим на сервер, переброс в браузе

git fetch
>покажет что есть изменения на сервере в файле
git status
>покажет что наша ветка отстает на N коммитов

master - локальная ветка
origin master(без /) ветка на сервере
origin/master - локальная но не рабочая, т.е. данные не совпадают между собой на локал и сервере

git pull - обновляем на локал изменения на сервере

cat назван_файла - смотрим содержимое файла

git log - просмотр изменений коммитов, которые проводились(кто, дата, какие изменения)
>q - для закрытия
>git log --author имя_автора - просмотр конкретного автора, что он делал

git show номер_коммита - что происходило с данным коммитом
>git show (без номера) - покажет последний коммит

git blame имя_файла - позволяет узнать об авторе строки файла
git blame имя_файла | grep 123(строки1,2,3)
git blame имя_файла | grep имя_автора

echo "Любой текст" >> имя_файла - добавляет в конец данного файла данный текст

git diff - показывает изменения как в git log и в git show
