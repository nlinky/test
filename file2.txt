git commit -m "test commit 1"
>m - месседж для коммита
> коммитим, название на сервере, но пока только у нас на локал
git commit -a -m "test commit 1"
git commit -am "test commit 1"

git reset HEAD~1 - вернуться на 1 шаг назад

git merge --abort - указывает где кофликт(смотреть в файле на локал)

git checkout название_файла - откатывает предыдущее действие
git checkout . - откатывает у всех последних файлов

git stash - откатывает изменение во врем хранилище
git stash pop - возвращает откатанное изменение
git stash clear - удаляет информацию из хранилища
----------создание ветки
git branch название_ветки - создание новой ветки

git branch - выводит все ветки
>* основная

git checkout название_ветки - переключиться на данную ветку
чтобы отправить на сервер, необходимо закомитить и запушить:

git push -u origin название_новой_ветки - отправить новую ветку на сервер

git checkout - для возвращения в ветку мастер

git merge название_ветки - замержить ветку в основную ветку

----работа с ветками--------
---переименование
git branch -m название_старой название_новой - переименование
>дальше нужно создать файл-коммит-пуш

git push origin :название_старой - для удаления старой т.к. она остается после переименования
и пушим новую переименованную ветку git push -u origin название_новой_ветки
---удаление
преходим в главную ветку 

git branch -d название_ветки - удаление с локал
>git push origin :название_старой - удаление с сервера
